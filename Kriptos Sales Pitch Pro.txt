        async function fetchWithRetry(url, options, maxRetries = 3) {
            const delays = [1000, 2000, 4000];
            for (let i = 0; i < maxRetries; i++) {
                try {
                    const response = await fetch(url, options);
                    if (!response.ok) {
                        throw new Error(`HTTP Error: ${response.status}`);
                    }
                    return await response.json();
                } catch (error) {
                    if (error.message.includes('401') || error.message.includes('400') || error.message.includes('404')) {
                        throw error; // No reintentar errores fatales de auth o formato
                    }
                    if (i === maxRetries - 1) throw error;
                    await new Promise(resolve => setTimeout(resolve, delays[i]));
                }
            }
        }

}`;

                // Se elimina responseMimeType para evitar el Error 400 que causa conflicto con google_search
                const payload = {
                    contents: [{ parts: [{ text: `Investiga en internet y genera el perfil B2B para la empresa: ${companyName}` }] }],
                    systemInstruction: { parts: [{ text: systemPrompt }] },
                    tools: [{ "google_search": {} }]

                if (!jsonText) throw new Error("La API no devolvió contenido válido.");
                
                // Extracción segura vía Regex por si la búsqueda inyecta texto extra o markdown
                const jsonMatch = jsonText.match(/\{[\s\S]*\}/);
                if (jsonMatch) {

                    errorMsg.innerHTML = "<strong>Error 404 (No Encontrado):</strong> Hubo un conflicto de modelos de IA. Revisa que tu API Key sea correcta.";
                } else if (err.message.includes("400")) {
                    errorMsg.innerHTML = "<strong>Error 400 (Bad Request):</strong> La IA rechazó la solicitud. Verifica el formato de la información.";
                } else if (err.message.includes("JSON")) {
                    errorMsg.innerHTML = "<strong>Error de Formato:</strong> La IA devolvió datos inconsistentes debido a la búsqueda web. Por favor, intenta de nuevo.";
                } else {
                    errorMsg.innerHTML = `<strong>Error inesperado:</strong> ${err.message}.`;

                loadingOverlay.classList.add('hidden');
                loadingOverlay.classList.remove('flex');
                btnText.innerText = "Investigar Web";
                btnIcon.innerText = "🌐";
                inputEl.disabled = false;
                document.getElementById('searchBtn').disabled = false;
            }
        }

            arrayData.forEach(item => { ul.innerHTML += htmlFormatFn(item); });
        }

        function toggleAccordion(id) {
            const content = document.getElementById(id);
            const iconId = id.replace('acc', 'icon');
            const icon = document.getElementById(iconId);
            
            if (content.style.maxHeight && content.style.maxHeight !== '0px') {
                content.style.maxHeight = '0px';
                icon.style.transform = 'rotate(0deg)';
            } else {
                content.style.maxHeight = content.scrollHeight + "px";
                icon.style.transform = 'rotate(180deg)';
            }
        }

        function renderChart(dataObj) {
            const ctx = document.getElementById('dataRiskChart').getContext('2d');
            
            if (currentChart) currentChart.destroy();
            if(!dataObj || !Array.isArray(dataObj)) return;

            const labels = dataObj.map(d => {
                const words = (d.label || '').split(' ');
                let lines = [], currentLine = words[0] || '';
                for (let i = 1; i < words.length; i++) {
                    if (currentLine.length + words[i].length + 1 <= 20) currentLine += " " + words[i];
                    else { lines.push(currentLine); currentLine = words[i]; }
                }
                lines.push(currentLine);
                return lines.length > 1 ? lines : lines[0];
            });
            
            const values = dataObj.map(d => d.value || 0);

            currentChart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: labels,
                    datasets: [{
                        data: values,
                        backgroundColor: ['#3b82f6', '#f59e0b', '#ef4444', '#64748b'],
                        borderWidth: 2, borderColor: '#ffffff', hoverOffset: 4
                    }]
                },
                options: {
                    responsive: true, maintainAspectRatio: false,
                    plugins: {
                        legend: { position: 'right', labels: { font: { family: "'Inter', sans-serif", size: 11 }, usePointStyle: true, padding: 15 } },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    let label = context.label || '';
                                    if (Array.isArray(label)) label = label.join(' ');
                                    return ` ${label}: ${context.parsed}%`;
                                }
                            },
                            backgroundColor: 'rgba(15, 23, 42, 0.9)', titleFont: { size: 13 }, bodyFont: { size: 12 }, padding: 10, cornerRadius: 4
                        }
                    },
                    cutout: '65%'
                }
            });
        }
